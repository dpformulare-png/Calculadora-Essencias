// Function to switch tabs function openTab(tabName) { var i, tabcontent, tablinks; tabcontent = document.getElementsByClassName("tabcontent"); for (i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; } tablinks = document.getElementsByClassName("tablinks"); for (i = 0; i < tablinks.length; i++) { tablinks[i].className = tablinks[i].className.replace(" active", ""); } document.getElementById(tabName).style.display = "block"; } // Function to update the rule function atualizarRegra() { var regra = document.getElementById("regraInput").value; document.getElementById("regraDisplay").innerText = regra; } // Function to calculate the base of the pyramid function calcularBase(height) { return height * height; } // Function to calculate pyramid volume function calcularPiramide(base, height) { return (1/3) * base * height; }
